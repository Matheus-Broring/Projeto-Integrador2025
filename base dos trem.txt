    <script>
        const toggle = document.getElementById('darkToggle');
        const lupaIcon = document.getElementById('lupaIcon');
        const logoMain = document.getElementById('logoMain');

        const logos = {
          default: {
            logoMain: '/style/img/image-Photoroom.png',
            lupa: '/style/img/lupa.png',
          },
          dark: {
            logoMain: '/style/img/image-Photoroom - Copia.png',
            lupa: '/style/img/lupa-B.png',
          },
        };

        const temaSalvo = localStorage.getItem('dark-mode');
        if (temaSalvo === 'true') {
          document.body.classList.add('dark-mode');
          toggle.checked = true;
          setDarkModeImages(true);
        }

        toggle.addEventListener('change', () => {
          const dark = toggle.checked;
          document.body.classList.toggle('dark-mode', dark);
          localStorage.setItem('dark-mode', dark);
          setDarkModeImages(dark);
        });

        function setDarkModeImages(isDark) {
          const theme = isDark ? logos.dark : logos.default;
          lupaIcon.src = theme.lupa;
          logoMain.src = theme.logoMain;
        }
    </script>
